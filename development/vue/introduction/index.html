<!-- 视图 HTML结构 -->
<!-- 插值表达式：支持数据/函数调用/简单表达式（加减乘除、三目运算符）--> 
<div id="app">
     {{a}}   {{ 5 < 4 ? "ok" : "wrong" }} {{ fn() }}  
</div>

<script src="./node_modules/vue/dist/vue.js"></script>

//2.创建vue实例

new Vue({
        el:"#app", //元素 不能使用body、html标签
        //数据      
        //1.对象方法
        data:{  
            a:100,
            b:200,
        },
    //2.函数方法
    data:function(){    
        return{
            a:100,
            b:200
        }
    }, 
    //3.函数简化方法(常用)
    data (){
        return{
            a:100,
            b:200
        }
    },
    methods:{
        fn(){
            return "hello vue.js"
        }
    }
});
</script>
