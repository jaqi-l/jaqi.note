# 7.8.1路由基础

  • 安装
npm install vue-router
  • 全局引用
    main.js
import VueRouter from 'vue-router'

Vue.use(VueRouter)
// 或
new Vue({
    render: h => h(App),
    router,
}).$mount('#app')
  • 使用

# js文件中：

    router.push('/index')

# vue文件中：

<router-link to="/index">首页</router-link> 
<!-- router-link被渲染成a元素，to属性被渲染成href属性 -->

 methods: {
    go() {
      this.$router.push('/index')
    },
    goBack(){
        this.$router.go(-1)// 后退一步记录，等同于 history.back()
    },
     goForward(){
        this.$router.go(1) // 在浏览器记录中前进一步，等同于 history.forward()
    }
  }

# 动态路由

  • query方法（查询路由）

# 编程式:

this.$router.push({ path: "/index", query: data })// 发送 -> /index
this.$route.query // 接收

# 声明式:

<router-link :to="{ path: '/news', query: data}">新闻</router-link>
  • params方法(命名路由)

# 编程式:

 this.$router.push({ name: "index", params: data })// 发送 -> /index?data
 tthis.$route.params // 接收

# 声明式:

<router-link :to="{ name: 'news', params: data}">新闻</router-link>
  • 路径传参

# 编程式:

this.$router.push({ path: "/index/:id"})

# 声明式:

<router-link to="/index/+'data'+">首页</router-link>

TIP

  1. query要用(path)路径传参，params要用(name)命名路由传参。
  2. 通过命名路由(params)传递的参数，刷新页面参数会丢失。
  3. 发送是$router,接收使是$route

# 路由捕获

 path: '*' // 会匹配所有路径，通常用于404错误
 path: '/user-*' // 会匹配以 `/user-` 开头的任意路径

# 路由嵌套

  routes: [
    { path: '/user/:id', component: User,
      children: [
        {
          // 当 /user/:id/profile 匹配成功，
          // UserProfile 会被渲染在 User 的 <router-view> 中
          path: 'profile',
          component: UserProfile
        },
        {
          // 当 /user/:id/posts 匹配成功
          // UserPosts 会被渲染在 User 的 <router-view> 中
          path: 'posts',
          children: []
        }
      ]
    }
  ]

TIP

以/开头的嵌套路径会被当作根路径

# 重定向与别名

# 从/a重定向到/b：

 routes: [
   { path: '/a', redirect: '/b' }
 ]

# 重定向另一个命名路由

 routes: [
  { path: '/a', redirect: { name: 'foo' }}
]

# 重定向一个方法

routes: [
  { path: '/a', redirect: to => {
    // 方法接收 目标路由 作为参数
    // return 重定向的 字符串路径/路径对象
  }}
]

# 别名

/a的别名是/b，意味着，当用户访问/b 时，URL会保持为 /b，但是路由匹配则为/a，就像用户访问 /a 一样。

routes: [
  { path: '/a', component: A, alias: '/b' }
]

# 路由组件传参

使组件只能在某些特定的 URL 上使用

# 取代与 $route 的耦合：

const User = {
  template: '<div>User {{ $route.params.id }}</div>'
}
const router = new VueRouter({
  routes: [
   { path: '/user/:id', component: User }
  ]
})

# 通过 props 解耦：

const User = {
    props: ['id'],
    template: '<div>User {{ id }}</div>'
}
const router = new VueRouter({
  routes: [
    { path: '/user/:id', component: User, props: true }, // props=true，route.params将会被设置为组件属性

    // 对于包含命名视图的路由，必须分别为每个命名视图添加 `props` 选项：
    {
      path: '/user/:id',
      components: { default: User, sidebar: Sidebar },
      props: { default: true, sidebar: false }
    }
  ]
})

# 7.8.2导航卫士

# 局部卫士：

beforeRouteEnter进入当前路由前
beforeRouteUpdate当前路由更新前
beforeRouteLeave离开当前路由前

# 全局卫士：

在main.js或router.js配置
beforeEach进入路由时
beforeResolve路由解析后
afterEach进入路由后，不接受next参数

# 独享卫士：

在router.js配置
beforeEnter进入路由配置时

# 参数：

from从哪个路由进入
to去哪个路由
next组件渲染完毕后回调

# 触发顺序：

beforeEach→beforeEnter→beforeRouteEnter→beforeRouteUpdate→beforeRouteLeave →beforeResolve→afterEach→beforeRouteEnter的回调函数

# 案例

  1. 守卫模块（作用于VueRouter）:
routes:[{
  path:"/Blog",
  component:Blog,
  meta:{
  auth:true// 表示用户访问该组件是否需要登陆
}]
  1. 登录模块（作用于登录模块）:
handleLogin(){   //保存用户名和密码
  localStorage.setItem("username",{name:this.uname,psw:this.psw});
  this.$router.push("/blog")}//给VueRouter添加一个历史记录 目的是渲染blog
}
  1. 守卫判断:
router.beforeEach((to,from,next)=>{
  if (to.meta.auth) {//判断有没有用户名和密码
      if (localStorage.getItem("username")) {//有用户名说明已经登录
        next();
      }else{//没有登录 渲染登录组件
        next({path:"/login"});
      };
    }else{
    next();
  };
});
  1. 注销模块（作用域实例）：
clearOut(){
  localStorage.removeItem("username"); // 移除本地用户名密码
  this.$router.push("/login") 
}//给VueRouter添加一个历史记录 目的是渲染blog
