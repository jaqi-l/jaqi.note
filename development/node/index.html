# 6.3node.js模块

http、fs、url....

项目复杂 1 命名冲突 2 文件依赖

一个js文件就是一个模块

1.系统核心模块：http、fs、url....

2.第三方模块：require("模块")

3.自定义模块：

将自定义模块放入node_modules文件夹中

require("./mod1.js");

补充：exports模块导出

exports与module.exports区别

1 每个模块都有module对象

2 module对象有一个exports对象

3 我们把需要导出的成员挂载到module.exports对象中

module.exports.*** = *** 完成可以用别名 exports.*** = ***

4 当一个模块导出单个成员 module.exports = XXX

总结 1 导出多个成员 推荐

exports.** = **

exports.** = **

      ....

  2 单个成员

module.exports = 成员

# 6.3.1响应请求（http）

http：服务器具备的基本功能，* 响应请求（http模块）

const http = require("http");

let server = http.createServer(function(req,res) { //一旦有请求过来 函数就执行

res.write("hello"); //向客户端发送信息

res.write("

hello

");

res.end(); //终止本次请求

});

server.listen(8000);//监听等待客户端的连接

# 6.3.5assert（断言）

const assert = require("assert");

assert(10>19,"达式3")//如果表达式错误则返回达式3

assert.equal(1,1)//比较两个值相不相等（不支持json）

assert.deepEqual(1,"1","达式3");//如果两个表达式不相等则返回表达式3，相当于==

assert.deepStrictEqual（1,"1","达式3"）//如果两个表达式不相等则返回表达式3，相当于===

# 6.3.6path

处理路径相关的

const path = require("path")

let pathUrl = "/a/b/c/1.html";

获取目录path.dirname(pathUrl);// /a/b/c ②获取扩展名path.extname(pathUrl);// .html

③获取基本名和扩展名path.basename(pathUrl);// 1.html

④拼接路径path.resolve("/root/a/b","../c","d","..","e");// /root/a/c/e（.当前目录 .. 上级 ...上上级）

# 6.4get/post

req.method 获取请求方式 GET/POST

# 6.4.1get

//①通过url模块：

//头部：
const url = require("url");

console.log(req.url); // /abc?username=abc&psw=123

let result = url.parse(req.url,true);

console.log(result.query.username);

console.log(result.query.psw);

//②通过querystring模块：

//头部：
const querystring = require("querystring");

 let query = req.url.split("?")[1]; // 截取？后面的username=abc&psw=123

 let uname = query.split("&")[0].split("=")[1];// 截取&后的username=abc，截取=后面的abc

console.log(uname);//abc

let res_get = querystring.parse(query); //{ username: 'abc',psw: '123' }

console.log(res_get.username,res_get.psw);//abc123

# 6.4.2post

//头部：
const querystring = require("querystring");

let arr = [];

req.on("data",function(buffer){ //每次有数据触发,调用回调函数

arr.push(buffer);//获取每段二进制数据

  });

req.on("end",function(){

    let buffer = Buffer.concat(arr);//合并二进制数据

    console.log(buffer.toString());//二进制数据转成字符串

    let post = querystring.parse(buffer.toString());//字符串转对象

    console.log(post);

  });
